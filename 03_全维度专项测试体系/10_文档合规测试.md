## 1. 需求文档测试

### 1.1 需求文档质量检查

| 质量属性 | 检查要点 | 典型问题 |
|----------|----------|----------|
| **正确性** | 需求描述与技术实现一致 | 文档描述的功能在实际系统中不存在 |
| **完整性** | 所有功能点都有描述，无遗漏 | 新加的功能没有更新需求文档 |
| **一致性** | 不同文档对同一功能描述一致 | PRD与API文档对同一个接口描述不同 |
| **清晰性** | 无歧义，不同人理解一致 | "用户登录失败后锁定账户"（未说明锁定多久） |
| **可测试性** | 每个需求都有明确的验收标准 | "系统应该快速响应"（未定义"快速"的具体数值） |
| **可追溯性** | 需求→设计→代码→测试用例可追溯 | 需求变更后相关测试用例未更新 |

### 1.2 需求评审Checklist

```
[ ] 需求是否明确描述了功能的前置条件和触发条件？
[ ] 异常场景是否被覆盖？（网络错误、数据为空、权限不足等）
[ ] 性能指标是否被量化？（响应时间<200ms，而非"快速"）
[ ] 安全需求是否被明确？（哪些数据需要加密、哪些操作需要审计）
[ ] 是否考虑了并发场景？
[ ] 是否考虑了数据量和增长趋势？
[ ] UI交互细节是否明确？（加载状态、空状态、错误状态）
[ ] 是否与已有功能存在冲突？
[ ] 验收标准是否可以客观验证？
[ ] 是否存在循环定义或逻辑矛盾？
```

```python
class TestRequirementTraceability:
    """需求可追溯性验证"""
    
    def test_requirements_all_implemented(self):
        """验证所有需求都已实现"""
        # 从需求管理系统获取所有需求
        requirements = self._get_all_requirements()
        
        unimplemented = []
        for req in requirements:
            # 查找对应的测试用例
            test_cases = self._find_test_cases_for_requirement(req['id'])
            if not test_cases:
                unimplemented.append(req)
                continue
            
            # 验证至少有一个通过的测试用例
            has_passing_test = any(
                tc['last_status'] == 'passed' for tc in test_cases
            )
            if not has_passing_test:
                unimplemented.append(req)
        
        if unimplemented:
            print(f"警告：{len(unimplemented)}个需求未被验证实现：")
            for req in unimplemented[:10]:
                print(f"  - [{req['id']}] {req['title']}")
        
        # 覆盖率应达到100%（或合理比例）
        coverage = 1 - len(unimplemented) / len(requirements)
        assert coverage >= 0.95, f"需求实现覆盖率仅{coverage:.1%}"
    
    def test_no_orphan_test_cases(self):
        """验证没有无对应需求的测试用例（可能是过度测试或需求遗漏）"""
        test_cases = self._get_all_test_cases()
        orphan_count = 0
        
        for tc in test_cases:
            linked_reqs = self._get_linked_requirements(tc['id'])
            if not linked_reqs:
                orphan_count += 1
        
        orphan_ratio = orphan_count / len(test_cases)
        assert orphan_ratio < 0.1, \
            f"无对应需求的测试用例过多：{orphan_count}/{len(test_cases)} ({orphan_ratio:.1%})"
```

```python
class TestDocumentation:
    """文档自动化验证"""
    
    def test_screenshots_match_ui(self):
        """验证文档截图与实际UI一致"""
        from PIL import Image
        import imagehash
        
        # 从文档中提取截图和对应的页面URL
        doc_screenshots = self._extract_screenshots_from_docs()
        
        mismatches = []
        for item in doc_screenshots:
            # 访问对应页面并截图
            actual_screenshot = self._take_screenshot(item['page_url'])
            
            # 计算感知哈希差异
            doc_hash = imagehash.phash(Image.open(item['image_path']))
            actual_hash = imagehash.phash(actual_screenshot)
            
            diff = doc_hash - actual_hash  # 汉明距离
            if diff > 5:  # 差异阈值（5/64≈8%差异）
                mismatches.append({
                    'page': item['page_url'],
                    'doc_image': item['image_path'],
                    'diff_score': diff,
                })
        
        assert len(mismatches) == 0, \
            f"文档中有{len(mismatches)}张截图与实际UI不一致"
    
    def test_operations_accuracy(self):
        """验证文档中描述的操作步骤可以正确执行"""
        from selenium import webdriver
        from selenium.webdriver.common.by import By
        
        driver = webdriver.Chrome()
        
        # 从文档中解析操作步骤
        documented_steps = self._parse_operation_steps("user_manual.html")
        
        errors = []
        for procedure_name, steps in documented_steps.items():
            try:
                for step in steps:
                    action = step['action']  # click, type, select, verify
                    target = step['target']  # CSS selector or xpath
                    value = step.get('value', '')
                    
                    if action == 'click':
                        element = driver.find_element(By.CSS_SELECTOR, target)
                        element.click()
                    elif action == 'type':
                        element = driver.find_element(By.CSS_SELECTOR, target)
                        element.send_keys(value)
                    elif action == 'verify':
                        element = driver.find_element(By.CSS_SELECTOR, target)
                        assert value in element.text
            except Exception as e:
                errors.append(f"{procedure_name}: {str(e)}")
        
        driver.quit()
        
        assert len(errors) == 0, \
            f"文档中有{len(errors)}个操作步骤无法复现"
    
    def test_term_consistency(self):
        """验证文档术语一致性"""
        import re
        from collections import defaultdict
        
        doc_content = self._load_document_text("user_manual.html")
        
        # 定义术语映射（同一个概念的所有指代）
        term_variants = {
            "登录": ["登录", "登入", "登陆", "Sign In", "Login"],
            "注销": ["注销", "登出", "退出登录", "Logout", "Sign Out"],
            "账户": ["账户", "帐号", "账号", "Account"],
            "设置": ["设置", "配置", "Settings", "Configuration"],
        }
        
        inconsistencies = []
        for canonical_term, variants in term_variants.items():
            found_variants = []
            for variant in variants:
                count = len(re.findall(re.escape(variant), doc_content))
                if count > 0:
                    found_variants.append((variant, count))
            
            # 如果发现了多个变体，说明术语不一致
            if len(found_variants) > 1:
                inconsistencies.append({
                    'canonical': canonical_term,
                    'variants_found': found_variants,
                })
        
        if inconsistencies:
            print("术语不一致问题：")
            for item in inconsistencies:
                print(f"  '{item['canonical']}'有多个变体：{item['variants_found']}")
```

### 3.1 主要法律法规/标准对比

| 法规/标准 | 适用地区 | 核心要求 | 违规后果 |
|-----------|----------|----------|----------|
| **GDPR** | 欧盟/处理欧盟公民数据 | 数据最小化、用户同意、被遗忘权、数据可携带、DPIA | 最高全球营收4%或2000万欧元 |
| **PIPL**（个人信息保护法） | 中国 | 告知同意、最小必要、单独同意（敏感信息）、数据本地化 | 最高5000万元或年营收5% |
| **等保2.0** | 中国 | 网络安全等级保护（5级）、技术+管理要求 | 警告、罚款、暂停业务 |
| **CCPA/CPRA** | 美国加州 | 消费者知情权、删除权、选择退出权（注：CPRA于2023年1月1日生效，是当前有效标准，修正并扩展了CCPA） | 每次违规$2500-$7500 |
| **HIPAA** | 美国（医疗） | PHI保护、最小必需原则、审计追踪 | 每次违规$100-$50000 |
| **PCI DSS** | 全球（支付卡行业） | 持卡人数据保护、加密传输、访问控制 | 罚款+吊销支付资质 |

## 4. 审计日志合规

### 4.1 审计日志要求

| 要求 | 说明 | 验证方法 |
|------|------|----------|
| **完整性** | 所有关键操作被记录 | 执行操作→验证日志包含该操作 |
| **不可篡改性** | 日志记录后不能被修改或删除 | 尝试修改日志→应被阻止或检测 |
| **可追溯性** | 每条日志可追溯到具体用户 | 验证日志中的用户标识、IP、时间戳 |
| **时序正确性** | 日志时间线正确 | 对比操作时间和日志时间 |
| **保留期限** | 满足法规要求的最低保留期 | 检查日志清理策略 |

## 5. 许可协议合规

### 5.1 开源许可协议兼容性

| 许可协议 | 类型 | 可以商用 | 需要开源 | 需要声明 | 典型组件 |
|----------|------|----------|----------|----------|----------|
| **MIT** | 宽松 | 是 | 否 | 是 | jQuery, React |
| **Apache 2.0** | 宽松 | 是 | 否 | 是 | Kubernetes, Spring |
| **BSD** | 宽松 | 是 | 否 | 是 | Go, Flask |
| **GPL** | Copyleft | 是 | 是（传染性） | 是 | MySQL, BusyBox |
| **AGPL** | 强Copyleft | 是 | 是（网络使用也触发） | 是 | MongoDB(旧版) |
| **LGPL** | 弱Copyleft | 是 | 仅修改库本身 | 是 | FFmpeg |
| **商业许可** | 商业 | 按条款 | 否 | 按条款 | 各种商业库 |

```python
def test_license_compliance():
    """许可证合规检查"""
    import subprocess
    import json
    
    # 1. 扫描所有依赖的许可证
    result = subprocess.run(
        ["npx", "license-checker", "--json", "--production"],
        capture_output=True, text=True
    )
    dependencies = json.loads(result.stdout)
    
    # 2. 检查不允许的许可证
    forbidden_licenses = [
        'GPL', 'AGPL', 'CC-BY-NC',  # 根据公司政策调整
    ]
    
    violations = []
    for pkg_name, pkg_info in dependencies.items():
        licenses = pkg_info.get('licenses', 'Unknown')
        if isinstance(licenses, str):
            licenses = [licenses]
        
        for lic in licenses:
            if any(forbidden in lic.upper() for forbidden in forbidden_licenses):
                violations.append({
                    'package': pkg_name,
                    'license': lic,
                    'version': pkg_info.get('version'),
                })
    
    # 3. 检查许可证文件是否存在
    license_file = Path("LICENSE") or Path("LICENSE.md")
    assert license_file.exists(), "项目根目录缺少LICENSE文件"
    
    # 4. 检查Notice文件（Apache 2.0要求）
    if 'Apache-2.0' in str(dependencies.values()):
        notice_file = Path("NOTICE")
        assert notice_file.exists(), \
            "使用了Apache 2.0组件但缺少NOTICE文件"
    
    if violations:
        print(f"发现{len(violations)}个许可证违规：")
        for v in violations:
            print(f"  {v['package']}@{v['version']} - {v['license']}")
    
    assert len(violations) == 0, f"存在{len(violations)}个许可证违规"
```

## 7. 本章小结

1. **文档测试不是"读一遍"**：需要结构化检查、自动化验证和可追溯性量化
2. **合规是法律底线**：GDPR/PIPL/等保的违规成本极高，合规测试是企业护城河
3. **审计日志不可忽视**：不仅要记录，更要保证完整、不可篡改、可追溯
4. **开源许可证是技术债**：使用前检查、定期审计、及时处理不兼容的许可证
5. **需求-设计-实现-文档的一致性**需要持续维护的工具体系
6. **自动化能大幅提高文档合规测试效率**：截图对比、链接检查、API一致性验证都应自动化
7. **合规不是一次性工作**：法规更新、系统变更、第三方依赖变化都可能影响合规状态
8. **文档合规测试要左移**：在设计阶段就考虑合规要求，而不是事后补救
