### 1.2 存储过程测试

```sql
-- 演示：订单创建的存储过程
DELIMITER $$
CREATE PROCEDURE sp_create_order(
    IN p_user_id INT,
    IN p_product_id INT,
    IN p_quantity INT,
    OUT p_order_id INT,
    OUT p_status_code INT
)
BEGIN
    DECLARE v_stock INT;
    DECLARE v_price DECIMAL(10,2);
    DECLARE EXIT HANDLER FOR SQLEXCEPTION
    BEGIN
        ROLLBACK;
        SET p_status_code = -1;
    END;
    
    START TRANSACTION;
    
    -- 获取商品价格和库存
    SELECT price, stock INTO v_price, v_stock 
    FROM products WHERE id = p_product_id FOR UPDATE;
    
    -- 检查库存
    IF v_stock < p_quantity THEN
        ROLLBACK;
        SET p_status_code = -2; -- 库存不足
    ELSE
        -- 创建订单
        INSERT INTO orders (user_id, status, total_amount) 
        VALUES (p_user_id, 'pending', v_price * p_quantity);
        SET p_order_id = LAST_INSERT_ID();
        
        -- 创建订单明细
        INSERT INTO order_items (order_id, product_id, quantity, price) 
        VALUES (p_order_id, p_product_id, p_quantity, v_price);
        
        -- 扣减库存
        UPDATE products SET stock = stock - p_quantity WHERE id = p_product_id;
        
        SET p_status_code = 0; -- 成功
    END IF;
    
    COMMIT;
END$$
DELIMITER ;
```

**存储过程测试用例**：

| 用例编号 | 场景 | 输入 | 预期输出 | 数据库状态验证 |

| SP-001 | 正常下单 | user=100, product=200, qty=2 | order_id>0, status=0 | orders+1行, order_items+1行, stock-2 |
| SP-002 | 库存不足 | user=100, product=201(stock=1), qty=5 | status_code=-2 | 无新记录, stock不变 |
| SP-003 | 商品不存在 | user=100, product=999, qty=1 | status_code=-1 | 无变化 |
| SP-004 | 数量为0 | user=100, product=200, qty=0 | status_code=-1或-2 | 无变化 |

```python
# 存储过程测试代码
import pytest
import pymysql
import threading

class TestStoreProcedures:
    @pytest.fixture
    def db(self):
        conn = pymysql.connect(
            host='localhost', user='test', password='test', 
            database='ecommerce_test'
        )
        yield conn
        conn.close()
    
    def test_sp_create_order_success(self, db):
        """测试正常下单"""
        with db.cursor() as cursor:
            # 先查询初始状态
            cursor.execute("SELECT stock FROM products WHERE id=200")
            initial_stock = cursor.fetchone()[0]
            
            # 调用存储过程
            cursor.callproc('sp_create_order', (100, 200, 2, 0, 0))
            db.commit()
            
            # 获取OUT参数
            cursor.execute("SELECT @_sp_create_order_3, @_sp_create_order_4")
            order_id, status_code = cursor.fetchone()
            
            assert order_id > 0, f"应返回有效的order_id，实际：{order_id}"
            assert status_code == 0, f"状态码应为0（成功），实际：{status_code}"
            
            # 验证订单创建
            cursor.execute("SELECT * FROM orders WHERE id=%s", (order_id,))
            order = cursor.fetchone()
            assert order is not None
            assert order[1] == 100  # user_id
            assert order[3] == 'pending'
            
            # 验证库存扣减
            cursor.execute("SELECT stock FROM products WHERE id=200")
            final_stock = cursor.fetchone()[0]
            assert final_stock == initial_stock - 2
    
    def test_sp_stock_insufficient(self, db):
        """测试库存不足场景"""
        with db.cursor() as cursor:
            cursor.callproc('sp_create_order', (100, 201, 99999, 0, 0))
            db.commit()
            cursor.execute("SELECT @_sp_create_order_4")
            status_code = cursor.fetchone()[0]
            assert status_code == -2, "库存不足应返回-2"
    
    def test_sp_concurrent_ordering(self, db):
        """测试并发下单 - 防止超卖"""
        errors = []
        
        def place_order():
            try:
                conn = pymysql.connect(
                    host='localhost', user='test', password='test',
                    database='ecommerce_test'
                )
                with conn.cursor() as cursor:
                    # 库存只有1个，两个并发请求各买1个
                    cursor.callproc('sp_create_order', (100, 202, 1, 0, 0))
                    conn.commit()
                    cursor.execute("SELECT @_sp_create_order_4")
                    status_code = cursor.fetchone()[0]
                    if status_code != 0:
                        errors.append(f"失败-状态码:{status_code}")
            except Exception as e:
                errors.append(str(e))
            finally:
                conn.close()
        
        # 并行执行
        threads = [threading.Thread(target=place_order) for _ in range(2)]
        for t in threads:
            t.start()
        for t in threads:
            t.join()
        
        # 验证只有一个成功
        success_count = 2 - len(errors)
        assert success_count == 1, f"并发下单应有且仅有1个成功，实际成功{success_count}个"
        
        # 验证库存为0（没有超卖）
        conn = pymysql.connect(
            host='localhost', user='test', password='test',
            database='ecommerce_test'
        )
        with conn.cursor() as cursor:
            cursor.execute("SELECT stock FROM products WHERE id=202")
            final_stock = cursor.fetchone()[0]
            assert final_stock == 0, f"最终库存应为0，实际：{final_stock}"
        conn.close()
```

## 2. 数据完整性测试

### 2.1 三大完整性

| 完整性类型 | 定义 | 数据库实现方式 | 测试方法 |
|-----------|------|--------------|----------|
| **实体完整性** | 每行数据有唯一标识，主键非空 | PRIMARY KEY, UNIQUE约束 | 尝试插入重复主键、NULL主键 |
| **参照完整性** | 外键值必须对应主表存在的主键值或为NULL | FOREIGN KEY, ON DELETE/UPDATE CASCADE | 尝试引用不存在的外键、删除被引用的记录 |
| **域完整性** | 列数据的类型、格式、取值范围符合定义 | CHECK约束, NOT NULL, DEFAULT, 数据类型 | 尝试插入超范围、错误类型、NULL值 |

```sql
-- 完整性测试用例
```

### 3.1 慢SQL分析


**常见慢SQL模式和优化方案**：

| 问题模式 | 示例 | 诊断方法 | 优化方案 |
|----------|------|----------|----------|
| **全表扫描** | `SELECT * FROM orders WHERE YEAR(created_at)=2025` | EXPLAIN查看type=ALL | 创建函数索引或改写WHERE条件 |
| **索引失效** | `SELECT * FROM users WHERE phone = 13800138000` (phone是varchar) | EXPLAIN查看type=ALL, key=NULL | 使用正确类型WHERE phone='13800138000' |
| **回表过多** | `SELECT * FROM orders WHERE user_id=1` | EXPLAIN查看Extra=Using where | 创建覆盖索引 |
| **大偏移量分页** | `SELECT * FROM orders LIMIT 500000, 20` | 执行时间随偏移量线性增长 | 使用游标分页 WHERE id>N LIMIT 20 |
| **N+1查询** | 循环中逐条查询 | 应用日志中大量重复查询 | 改为JOIN或批量查询 |
| **锁等待** | `SELECT ... FOR UPDATE` 持有锁时间过长 | SHOW ENGINE INNODB STATUS | 缩小事务范围、优化查询 |
| **排序无索引** | `SELECT ... ORDER BY created_at LIMIT 20` | EXPLAIN Extra=Using filesort | 创建(created_at)索引 |

### 3.2 执行计划分析(EXPLAIN)

```sql
-- EXPLAIN解读
EXPLAIN SELECT o.*, u.username 
FROM orders o 
JOIN users u ON o.user_id = u.id 
WHERE o.created_at > '2025-01-01' AND o.status = 'paid'
ORDER BY o.created_at DESC
LIMIT 20;
```

| EXPLAIN字段 | 含义 | 好的值 | 坏的值 |
|-------------|------|--------|--------|
| **type** | 访问类型（性能从好到差） | const, eq_ref, ref, range | ALL（全表扫描）, index（索引全扫描） |
| **key** | 实际使用的索引 | 有索引名 | NULL（未使用索引） |
| **rows** | 预估扫描行数 | 越小越好 | 接近表总行数 |
| **Extra** | 额外信息 | Using index（覆盖索引） | Using filesort, Using temporary, Using where |
| **filtered** | 过滤后剩余行百分比 | 越高越好 | 很低（大量数据在服务端过滤） |

**优化前后对比示例**：

```sql
-- 优化前：全表扫描
EXPLAIN SELECT * FROM orders WHERE DATE(created_at) = '2025-01-15';
-- type=ALL, key=NULL, rows=10000000

-- 优化后：使用索引范围扫描
CREATE INDEX idx_orders_created_at ON orders(created_at);
EXPLAIN SELECT * FROM orders 
WHERE created_at >= '2025-01-15 00:00:00' AND created_at < '2025-01-16 00:00:00';
-- type=range, key=idx_orders_created_at, rows=5000
```

```python
class TestIndexOptimization:
    """索引优化对比测试"""
    
    def setup_method(self):

        with self.conn.cursor() as cursor:

                CREATE TABLE IF NOT EXISTS perf_orders (
                    id BIGINT AUTO_INCREMENT PRIMARY KEY,
                    user_id INT NOT NULL,
                    status VARCHAR(20),
                    created_at DATETIME,
                    amount DECIMAL(10,2)
                ) ENGINE=InnoDB
            """)
            # 插入100万行测试数据
            # ...
    
    def test_query_with_and_without_index(self):
        """对比有无索引的查询性能"""
        
        # 1. 无索引查询
        start = time.time()
        with self.conn.cursor() as cursor:
            cursor.execute("SELECT * FROM perf_orders WHERE user_id = 50000")
            cursor.fetchall()
        time_without_index = time.time() - start
        
        # 2. 创建索引
        with self.conn.cursor() as cursor:
            cursor.execute("CREATE INDEX idx_user_id ON perf_orders(user_id)")
        
        # 3. 有索引查询
        start = time.time()
        with self.conn.cursor() as cursor:
            cursor.execute("SELECT * FROM perf_orders WHERE user_id = 50000")
            cursor.fetchall()
        time_with_index = time.time() - start
        
        # 4. 验证索引效果
        speedup = time_without_index / time_with_index
        assert speedup > 10, \
            f"索引加速比仅为{speedup:.1f}x，预期至少10x"
        
        print(f"无索引：{time_without_index:.4f}s")
        print(f"有索引：{time_with_index:.4f}s")
        print(f"加速比：{speedup:.1f}x")
    
    def test_covering_index_vs_regular_index(self):
        """对比覆盖索引和普通索引的性能差异"""
        # 普通索引：idx_user_id (user_id)
        # 覆盖索引：idx_user_id_status_amount (user_id, status, amount)
        
        query = """
            SELECT user_id, status, amount 
            FROM perf_orders 
            WHERE user_id BETWEEN 10000 AND 20000
        """
        
        # 使用覆盖索引时的Extra应为"Using index"
        with self.conn.cursor() as cursor:
            cursor.execute("EXPLAIN " + query)
            explain = cursor.fetchall()
            # 检查Extra是否包含Using index
            extra_info = explain[0][6] if len(explain[0]) > 6 else ''
            assert 'Using index' in str(extra_info), \
                f"覆盖索引应显示Using index，实际Extra: {extra_info}"
```

```python
class TestDataMigration:
    """数据迁移验证"""
    
    def test_row_count_consistency(self):
        """验证源库和目标库行数一致"""
        tables = ['users', 'orders', 'order_items', 'products', 'payments']
        
        for table in tables:
            source_count = self._get_row_count(self.source_db, table)
            target_count = self._get_row_count(self.target_db, table)
            assert source_count == target_count, \
                f"表 {table} 行数不一致：源={source_count}, 目标={target_count}"
    
    def test_data_integrity_sampling(self):
        """抽样验证数据内容一致性"""
        sample_size = 1000
        
        for table in ['users', 'orders', 'products']:
            # 随机抽样
            source_samples = self._random_sample(self.source_db, table, sample_size)
            
            for row in source_samples:
                target_row = self._get_row_by_id(self.target_db, table, row['id'])
                
                # 逐字段比较
                for col, value in row.items():
                    target_value = target_row.get(col)
                    if isinstance(value, Decimal):
                        assert abs(float(value) - float(target_value)) < 0.01, \
                            f"表{table} ID={row['id']} 字段{col}不一致"
                    elif isinstance(value, datetime):
                        # 允许秒级精度差异
                        diff = abs((value - target_value).total_seconds())
                        assert diff < 1, \
                            f"表{table} ID={row['id']} 字段{col}时间不一致"
                    else:
                        assert value == target_value, \
                            f"表{table} ID={row['id']} 字段{col}: {value} != {target_value}"
    
    def test_checksum_consistency(self):
        """使用CHECKSUM验证表级一致性"""
        tables = ['users', 'orders', 'order_items', 'products']
        
        for table in tables:
            source_checksum = self._get_table_checksum(self.source_db, table)
            target_checksum = self._get_table_checksum(self.target_db, table)
            assert source_checksum == target_checksum, \
                f"表 {table} CHECKSUM不一致"
    
    def test_business_rules_after_migration(self):
        """验证迁移后业务规则仍然正确"""
        # 检查外键关系
        orphan_records = self._find_orphan_foreign_keys(self.target_db)
        assert len(orphan_records) == 0, \
            f"迁移后存在{len(orphan_records)}条孤儿外键记录"
        
        # 检查唯一约束
        duplicate_records = self._find_duplicates(self.target_db)
        assert len(duplicate_records) == 0, \
            f"迁移后存在{len(duplicate_records)}条重复记录"
```

### 5.1 主从复制一致性

| 测试场景 | 测试方法 | 验证点 |
|----------|----------|--------|
| **正常同步延迟** | 写入主库后查询从库 | Seconds_Behind_Master < 阈值（如5秒） |
| **大量写入同步** | 批量插入10万条后查从库 | 从库延迟在接受范围内 |
| **网络中断恢复** | 断开从库网络5分钟→恢复 | 自动追上，数据无丢失 |
| **主从切换** | 执行主从切换 | 无数据丢失，新主库可写入 |

```python
def test_master_slave_consistency():
    """主从数据一致性测试"""
    import time
    
    # 1. 在主库写入
    master_conn = pymysql.connect(host='master-db', user='test', password='test')
    with master_conn.cursor() as cursor:
        cursor.execute("INSERT INTO test_replication (data) VALUES ('sync_test')")
        master_conn.commit()
        inserted_id = cursor.lastrowid
    
    # 2. 在从库读取（等待复制）
    slave_conn = pymysql.connect(host='slave-db', user='test', password='test')
    
    max_wait = 10  # 最长等待10秒
    start = time.time()
    synced = False
    
    while time.time() - start < max_wait:
        # 检查复制延迟
        with slave_conn.cursor() as cursor:
            # MySQL 8.0+推荐使用SHOW REPLICA STATUS替代已弃用的SHOW SLAVE STATUS
            cursor.execute("SHOW REPLICA STATUS")
            slave_status = cursor.fetchone()
            seconds_behind = slave_status[32] if slave_status else None
            
            if seconds_behind is not None and seconds_behind == 0:
                # 延迟为0，验证数据
                cursor.execute(
                    "SELECT data FROM test_replication WHERE id=%s", 
                    (inserted_id,)
                )
                row = cursor.fetchone()
                if row and row[0] == 'sync_test':
                    synced = True
                    break
        time.sleep(0.5)
    
    assert synced, f"从库在{max_wait}秒内未同步到数据"
    master_conn.close()
    slave_conn.close()
```

## 7. 本章小结

1. **数据库测试是多层次的**：从CRUD功能到数据完整性，再到性能、迁移和一致性
2. **数据完整性是底线**：实体、参照、域三大完整性应作为每轮回归的必测项
3. **性能测试需要真实数据量**：小数据集上的SQL优化效果不能代表生产环境
4. **EXPLAIN是必备技能**：每个开发和测试人员都应该能读懂执行计划
5. **迁移测试要有回退方案**：每次数据迁移都必须验证回退方案
6. **一致性验证要有量化指标**：延迟窗口、数据差异容忍度需要明确定义
7. **自动化是必须的**：数据校验SQL应集成到CI/CD中自动执行
8. **备份恢复要定期演练**：不是备份了就安全了，要证明能恢复
