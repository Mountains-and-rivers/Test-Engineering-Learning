## 案例1：Web端UI自动化测试框架从0到1搭建

### 1.1 项目背景

**项目名称**：某中型电商平台Web端UI自动化框架建设

**项目现状**：
- 前端技术栈：Vue.js 3.x + Element Plus
- 核心页面模块：首页、商品详情、购物车、下单、个人中心、商家后台（共约120个页面）
- 迭代周期：双周迭代，每次迭代约15-30个页面发生变更
- 测试团队：8人，其中2人有自动化经验
- 回归测试现状：每次发版前手工回归5天，加班严重，漏测率约8%

**立项动机**：
1. 回归测试耗时5天，占整个测试周期60%，影响发版节奏
2. 手工回归重复性高，人员疲劳导致漏测
3. 核心业务流程（下单、支付）需要每次发版100%验证
4. 浏览器兼容性测试（Chrome/Firefox/Edge/Safari）手工执行效率低

### 1.3 核心代码实现

#### 1.3.1 基础页面对象

```python
from playwright.sync_api import Page, Locator, expect
from typing import Optional
import allure

class BasePage:
    """所有页面对象的基类"""
    
    def __init__(self, page: Page):
        self.page = page
        self.timeout = 30000  # 默认30秒超时
    
    def navigate(self, url: str):
        """导航到指定URL"""
        with allure.step(f"导航到: {url}"):
            self.page.goto(url, wait_until="networkidle")
    
    def wait_for_element(self, selector: str, timeout: Optional[int] = None) -> Locator:
        """等待元素可见"""
        timeout = timeout or self.timeout
        locator = self.page.locator(selector)
        locator.wait_for(state="visible", timeout=timeout)
        return locator
    
    def click(self, selector: str) -> None:
        """点击元素"""
        with allure.step(f"点击: {selector}"):
            self.wait_for_element(selector)
            self.page.locator(selector).click()
    
    def fill(self, selector: str, text: str) -> None:
        """输入文本"""
        with allure.step(f"输入: {text} 到 {selector}"):
            self.wait_for_element(selector)
            self.page.locator(selector).fill(text)
    
    def get_text(self, selector: str) -> str:
        """获取元素文本"""
        return self.page.locator(selector).text_content()
    
    def take_screenshot(self, name: str) -> None:
        """截图并附加到Allure报告"""
        screenshot = self.page.screenshot(full_page=True)
        allure.attach(screenshot, name=name, 
                      attachment_type=allure.attachment_type.PNG)
    
    def assert_url_contains(self, expected: str) -> None:
        """断言当前URL包含指定字符串"""
        expect(self.page).to_have_url(lambda url: expected in url)
```

#### 1.3.2 浏览器管理工厂

```python
from playwright.sync_api import sync_playwright, Browser, BrowserContext, Page
from typing import Dict, Optional
import pytest

class BrowserFactory:
    """浏览器工厂 - 支持多浏览器类型和配置"""
    
    BROWSER_CONFIGS = {
        "chrome": {
            "channel": "chrome",
            "args": ["--disable-gpu", "--no-sandbox"],
            "viewport": {"width": 1920, "height": 1080}
        },
        "firefox": {
            "channel": None,  # 使用Playwright内置Firefox
            "viewport": {"width": 1920, "height": 1080}
        },
        "mobile_chrome": {
            "channel": "chrome",
            "viewport": {"width": 375, "height": 812},
            "is_mobile": True,
            "has_touch": True,
            "user_agent": "Mozilla/5.0 (iPhone; CPU iPhone OS 15_0 like Mac OS X)"
        }
    }
    
    def __init__(self, browser_type: str = "chrome"):
        self.browser_type = browser_type
        self.config = self.BROWSER_CONFIGS.get(browser_type, self.BROWSER_CONFIGS["chrome"])
        self._playwright = None
        self._browser = None
    
    def start(self) -> Page:
        """启动浏览器并返回Page对象"""
        self._playwright = sync_playwright().start()
        
        launch_options = {"headless": True, "slow_mo": 0}
        if "args" in self.config:
            launch_options["args"] = self.config["args"]
        if "channel" in self.config and self.config["channel"]:
            launch_options["channel"] = self.config["channel"]
        
        self._browser = self._playwright.chromium.launch(**launch_options)
        
        context_options = {k: v for k, v in self.config.items() 
                          if k in ["viewport", "is_mobile", "has_touch", "user_agent"]}
        context = self._browser.new_context(**context_options)
        
        # 注入全局设置
        context.set_default_timeout(30000)
        
        return context.new_page()
    
    def stop(self):
        """关闭浏览器"""
        if self._browser:
            self._browser.close()
        if self._playwright:
            self._playwright.stop()
```

### 1.5 CI/CD集成

```groovy
// Jenkins Pipeline
pipeline {
    agent { label 'docker-runner' }
    
    parameters {
        choice(name: 'BROWSER', choices: ['chrome', 'firefox', 'all'])
        choice(name: 'TEST_TYPE', choices: ['smoke', 'regression', 'full'])
        string(name: 'ENV', defaultValue: 'staging')
    }
    
    stages {
        stage('Checkout') {
            steps { checkout scm }
        }
        
        stage('Build Docker Image') {
            steps {
                sh 'docker build -t ui-test-runner:${BUILD_ID} -f Dockerfile.test .'
            }
        }
        
        stage('Smoke Tests') {
            when { expression { params.TEST_TYPE in ['smoke', 'regression', 'full'] } }
            steps {
                sh '''
                    docker run --rm \
                        -e BROWSER=${BROWSER} \
                        -e ENV=${ENV} \
                        -v ${WORKSPACE}/allure-results:/app/allure-results \
                        ui-test-runner:${BUILD_ID} \
                        pytest -m smoke --alluredir=allure-results
                '''
            }
        }
        
        stage('Regression Tests') {
            when { expression { params.TEST_TYPE in ['regression', 'full'] } }
            parallel {
                stage('Order Module') {
                    steps {
                        sh 'docker run --rm ... pytest test_order/ -n 4 --alluredir=allure-results'
                    }
                }
                stage('Product Module') {
                    steps {
                        sh 'docker run --rm ... pytest test_product/ -n 4 --alluredir=allure-results'
                    }
                }
                stage('Cart Module') {
                    steps {
                        sh 'docker run --rm ... pytest test_cart/ -n 3 --alluredir=allure-results'
                    }
                }
            }
        }
        
        stage('Generate Report') {
            steps {
                allure includeProperties: false,
                       results: [[path: 'allure-results']]
            }
        }
    }
    
    post {
        failure {
            emailext (
                subject: "[自动化测试失败] ${JOB_NAME} #${BUILD_NUMBER}",
                body: "测试执行失败，请检查Allure报告: ${BUILD_URL}allure",
                to: "qa-team@company.com"
            )
        }
    }
}
```

```python
        """创建Appium Driver"""
        if self.config["platformName"] == "Android":

        else:
            options = XCUITestOptions().load_capabilities(self.config)
```

```python
                element = self.find_by_text(text)
```

```python
    INCREMENTAL = "incr"   # 增量比对
    AGGREGATED = "agg"     # 聚合比对
    SAMPLE = "sample"      # 抽样比对

@dataclass
class CompareResult:
    """比对结果"""
    task_name: str
    source_a_count: int
    source_b_count: int
    matched_count: int
    diff_count: int
    only_in_a_count: int
    only_in_b_count: int
    diffs: List[Dict] = field(default_factory=list)
    execution_time_seconds: float = 0
    
    @property
    def match_rate(self) -> float:
        total = max(self.source_a_count, self.source_b_count)
        return self.matched_count / total if total > 0 else 1.0
    
    @property
    def passed(self) -> bool:
        return self.match_rate >= 0.9999  # 99.99%一致率

class DataCompareEngine:
    """数据比对引擎"""
    
    def __init__(self, source_a_conn, source_b_conn):
        self.conn_a = source_a_conn
        self.conn_b = source_b_conn
    
    def compare_full(self, df_a: pd.DataFrame, df_b: pd.DataFrame,
                     config: Dict) -> CompareResult:
        """全量比对"""
        key_cols = config.get("key_columns", [])
        compare_cols = config.get("compare_columns", [])
        
        # 设置索引
        df_a_idx = df_a.set_index(key_cols)
        df_b_idx = df_b.set_index(key_cols)
        
        result = CompareResult(
            task_name=config.get("task_name", "unknown"),
            source_a_count=len(df_a),
            source_b_count=len(df_b)
        )
        
        # 找出仅在A或仅在B的记录
        keys_a = set(df_a_idx.index)
        keys_b = set(df_b_idx.index)
        common_keys = keys_a & keys_b
        
        result.only_in_a_count = len(keys_a - keys_b)
        result.only_in_b_count = len(keys_b - keys_a)
        
        # 逐条比对共同记录
        diffs = []
        for key in common_keys:
            row_a = df_a_idx.loc[key]
            row_b = df_b_idx.loc[key]
            
            for col_config in compare_cols:
                col = col_config["column"]
                col_type = col_config.get("type", "exact")
                
                if not self._compare_value(row_a[col], row_b[col], col_config):
                    diffs.append({
                        "key": str(key),
                        "column": col,
                        "value_a": str(row_a[col]),
                        "value_b": str(row_b[col])
                    })
        
        result.diffs = diffs
        result.diff_count = len(diffs)
        result.matched_count = len(common_keys) * len(compare_cols) - len(diffs)
        
        return result
    
    def _compare_value(self, val_a, val_b, col_config: Dict) -> bool:
        """单个值比对"""
        col_type = col_config.get("type", "exact")
        
        if pd.isna(val_a) and pd.isna(val_b):
            return True
        
        if col_type == "exact":
            return str(val_a).strip() == str(val_b).strip()
        
        elif col_type == "numeric":
            tolerance = col_config.get("tolerance", 0)
            if pd.isna(val_a) or pd.isna(val_b):
                return False
            return abs(float(val_a) - float(val_b)) <= tolerance
        
        elif col_type == "datetime":
            tolerance_sec = col_config.get("tolerance_seconds", 0)
            diff = abs(pd.Timestamp(val_a) - pd.Timestamp(val_b))
            return diff.total_seconds() <= tolerance_sec
        
        return False
```
