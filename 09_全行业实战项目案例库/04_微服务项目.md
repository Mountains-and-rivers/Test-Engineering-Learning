- 微服务框架：Spring Cloud Hoxton.SR12
- 注册中心：Nacos 2.x（AP模式）
- 配置中心：Nacos Config
- 服务调用：OpenFeign + Sentinel（熔断降级）

- 消息队列：RocketMQ（事务消息）
- 分布式事务：Seata AT模式

```java
@SpringBootTest(
    webEnvironment = SpringBootTest.WebEnvironment.RANDOM_PORT,
    classes = {OrderServiceApplication.class}
)
@TestMethodOrder(MethodOrderer.OrderAnnotation.class)
@AutoConfigureMockMvc
class OrderControllerComponentTest {

    @Autowired
    private TestRestTemplate restTemplate;
    
    @MockBean  // Mock外部服务调用
    private UserServiceFeignClient userServiceFeignClient;
    
    @MockBean  // Mock外部服务调用
    private InventoryServiceFeignClient inventoryServiceFeignClient;
    
    @MockBean  // Mock消息队列
    private RocketMQTemplate rocketMQTemplate;
    
    // 使用H2内存数据库替代MySQL（数据库层面的隔离）
    @Autowired
    private DataSource dataSource;
    
    @BeforeEach
    void setUp() {
        // 初始化测试数据
        JdbcTemplate jdbc = new JdbcTemplate(dataSource);
        jdbc.execute("INSERT INTO test_data (...) VALUES (...)");
    }
    
    @Test
    @DisplayName("TC_ORDER_001: 正常创建订单")
    void testCreateOrder_Success() {
        // Arrange: Mock用户服务返回
        UserDTO mockUser = UserDTO.builder()
            .id(10001L).name("测试用户")
            .status("ACTIVE").build();
        when(userServiceFeignClient.getUserById(10001L))
            .thenReturn(Response.success(mockUser));
        
        // Arrange: Mock库存服务返回
        InventoryDTO mockInventory = InventoryDTO.builder()
            .skuId("SKU001").stock(100).build();
        when(inventoryServiceFeignClient.queryInventory(any()))
            .thenReturn(Response.success(mockInventory));
        
        // Act: 调用订单创建接口
        CreateOrderRequest request = CreateOrderRequest.builder()
            .userId(10001L)
            .items(List.of(OrderItem.builder()
                .skuId("SKU001").quantity(2).build()))
            .addressId(20001L)
            .build();
        
        ResponseEntity<ApiResponse<OrderDTO>> response = 
            restTemplate.postForEntity(
                "/api/v1/orders",
                request,
                new ParameterizedTypeReference<>() {}
            );
        
        // Assert
        assertThat(response.getStatusCode()).isEqualTo(HttpStatus.CREATED);
        assertThat(response.getBody().getCode()).isEqualTo("200");
        assertThat(response.getBody().getData().getStatus())
            .isEqualTo(OrderStatus.PENDING_PAYMENT);
        assertThat(response.getBody().getData().getTotalAmount())
            .isEqualByComparingTo(new BigDecimal("398.00"));
        
        // Verify: 验证外部调用
        verify(inventoryServiceFeignClient, times(1))
            .lockInventory(any(InventoryLockRequest.class));
        verify(rocketMQTemplate, times(1))
            .send(eq("ORDER_CREATED_TOPIC"), any(Message.class));
    }
    
    @Test
    @DisplayName("TC_ORDER_002: 库存不足创建订单失败")
    void testCreateOrder_InsufficientStock() {
        // Arrange: 库存不足
        InventoryDTO mockInventory = InventoryDTO.builder()
            .skuId("SKU001").stock(1).build();  // 库存仅1件
        when(inventoryServiceFeignClient.queryInventory(any()))
            .thenReturn(Response.success(mockInventory));
        
        // Act: 请求购买2件
        CreateOrderRequest request = CreateOrderRequest.builder()
            .userId(10001L)
            .items(List.of(OrderItem.builder()
                .skuId("SKU001").quantity(2).build()))  // 要买2件
            .build();
        
        ResponseEntity<ApiResponse<OrderDTO>> response = 
            restTemplate.postForEntity("/api/v1/orders", request, ...);
        
        // Assert
        assertThat(response.getBody().getCode()).isEqualTo("INSUFFICIENT_STOCK");
    }
}
```

```java
        DockerImageName.parse("apache/rocketmq:5.1.0-alpine"))

        .dependsOn(rocketmqNamesrv);
```

#### 2.3.2 健康检查测试示例

```yaml
# 健康检查配置
apiVersion: apps/v1
kind: Deployment
metadata:
  name: order-service
spec:
  template:
    spec:
      containers:
      - name: order-service
        image: order-service:latest
        ports:
        - containerPort: 8080
        # 就绪探针: 服务是否可接收流量
        readinessProbe:
          httpGet:
            path: /actuator/health/readiness
            port: 8080
          initialDelaySeconds: 30    # 启动后30秒开始检查
          periodSeconds: 10          # 每10秒检查一次
          timeoutSeconds: 5          # 超时5秒
          failureThreshold: 3        # 连续失败3次标记为未就绪
          successThreshold: 1        # 成功1次即标记为就绪
        
        # 存活探针: 服务是否需要重启
        livenessProbe:
          httpGet:
            path: /actuator/health/liveness
            port: 8080
          initialDelaySeconds: 60
          periodSeconds: 30
          timeoutSeconds: 5
          failureThreshold: 5        # 连续失败5次重启Pod
        
        # 启动探针: 只在启动阶段使用（K8s 1.18+）
        startupProbe:
          httpGet:
            path: /actuator/health
            port: 8080
          initialDelaySeconds: 10
          periodSeconds: 5
          failureThreshold: 30       # 最多等待150秒启动
```

#### 5.3.2 实验定义示例

```yaml
# LitmusChaos 实验定义 - Pod Kill
apiVersion: litmuschaos.io/v1alpha1
kind: ChaosEngine
metadata:
  name: order-service-pod-kill
  namespace: test
spec:
  appinfo:
    appns: test
    applabel: "app=order-service"
    appkind: deployment
  chaosServiceAccount: litmus-admin
  experiments:
    - name: pod-delete
      spec:
        components:
          # 渐进式增加爆炸半径
          - name: pod-delete
            env:
              - name: TOTAL_CHAOS_DURATION
                value: "60"           # 实验持续60秒
              - name: CHAOS_INTERVAL
                value: "10"           # 每10秒Kill一次
              - name: PODS_AFFECTED_PERC
                value: "30"           # 影响30%的Pod (爆炸半径)
              - name: TARGET_CONTAINER
                value: "order-service"
              - name: LIB_IMAGE
                value: "litmuschaos/go-runner:latest"
        probe:
          # 稳态验证探针 - 实验期间持续验证
          - name: "check-order-api-availability"
            type: "httpProbe"
            httpProbe/inputs:
              url: "http://order-service.test:8080/actuator/health"
              method:
                get:
                  criteria: "=="
                  responseCode: "200"
            mode: "Continuous"
            runProperties:
              probeTimeout: 5
              interval: 2
              retry: 1
              probePollingInterval: 1
              
      # 终止条件
      - name: "abort-on-high-error-rate"
        type: "promProbe"
        promProbe/inputs:
          endpoint: "http://prometheus:9090"
          query: |
            rate(http_requests_total{status=~"5.."}[1m]) /
            rate(http_requests_total[1m]) > 0.05
          comparator:
            type: "Float"
            criteria: ">="
            value: "0.05"        # 错误率 > 5% 自动终止实验
```

### 5.5 混沌工程自动化流水线

```yaml
# GitLab CI - 混沌实验自动执行
chaos-experiment:
  stage: chaos_test
  image: litmuschaos/chaos-operator:latest
  script:
    # 1. 部署实验定义
    - kubectl apply -f chaos/experiments/${EXPERIMENT_NAME}.yaml
    
    # 2. 等待实验完成
    - kubectl wait --for=condition=complete chaosengine/${EXPERIMENT_NAME} 
      --timeout=600s -n chaos
    
    # 3. 获取实验结果
    - kubectl get chaosresult ${EXPERIMENT_NAME} -n chaos -o json > result.json
    
    # 4. 验证结果
    - |
      VERDICT=$(jq -r '.status.experimentStatus.verdict' result.json)
      if [ "$VERDICT" != "Pass" ]; then
        echo "混沌实验失败: $VERDICT"
        cat result.json
        exit 1
      fi
      echo "混沌实验通过"
    
    # 5. 归档结果
    - cp result.json chaos-results/${EXPERIMENT_NAME}_$(date +%Y%m%d).json
  artifacts:
    paths:
      - chaos-results/
    when: always
  rules:
    - if: $CI_PIPELINE_SOURCE == "schedule"  # 定时触发
  allow_failure: true  # 混沌实验失败不阻塞流水线
```
